union List begin
  Nil[];
  Cons[Promise*(Int), List]
end

func writeToAll(l: List, v: Int): Unit begin
  match l begin
    Nil[] -> ()
    Cons[p, ll] ->
      p <- v;
      writeToAll(ll, v + 1)
  end
end

func buildList(v: Int): List begin
  if v == 0 then
    Nil[]
  else
    promise r, w: Int in
    Cons[w, buildList(v - 1)]
  end
end

func main(): Unit begin
  promise r, w: Int in
  w <- 5;
  if ?r == 5 then
    writeToAll(buildList(10), 0)
  else
    writeToAll(buildList(5), 0)
  end
end
