  for i = 0 to n begin
    ()
  end
end

  waitN(1000000);
  555
end

func future(): Promise(Int) begin
  promise r, w: Int in
  async w <- task();
  r
end

  addInt(n, n)
end

  let result: Promise(Int) = future() in
