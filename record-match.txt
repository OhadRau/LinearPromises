union IntList begin
  Empty[];
  Node[Int, IntList]
end

record PairOfLists begin
  left: IntList;
  right: IntList
end

func eq(a: Int, b: Int): Bool begin
  true
end

func main(): Unit begin
  let u: IntList = Empty[] in
  let p: PairOfLists = PairOfLists { left=u, right=u } in
  match p begin
    { left = l, right = r } ->
      match l begin
        Empty[] -> ()
        Node[a, b] ->
          if eq(a, 5) then () else () end
      end;
      match r begin
        Empty[] -> ()
        Node[a, b] ->
          if eq(a, 5) then () else () end
      end
  end
end
